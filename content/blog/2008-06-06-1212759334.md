<p><pre>#include<stdio.h><br />
#include<math.h><br />
#define N 4</p><p>main(){<br />
	double a[N][N],b[N],ell,a_max;<br />
	int pivot[N],i,j,k,m,m_opt,k_step_old;<br />
    <br />
	//初期設定<br />
	a[0][0]= 1.0; a[0][1]= 1.0; a[0][2]= 1.0; a[0][3]= 1.0;<br />
	a[1][0]= 1.0; a[1][1]= 1.0; a[1][2]= 2.0; a[1][3]= 2.0;<br />
	a[2][0]= 1.0; a[2][1]= 2.0; a[2][2]= -1.0; a[2][3]= 2.0;<br />
	a[3][0]= 1.0; a[3][1]= 2.0; a[3][2]= 2.0; a[3][3]= -1.0;</p><p>	b[0]= 1.0; b[1]= 2.0; b[2]= 1.0; b[3]= 2.0; </p><p>    //仮のピボット選択<br />
    for(i=0;i<N;i++){
        pivot[i] = i;
    }

    for(k=0;k<N-1;k++){
        /*k段以降の中から|a[pivot[m]][k]|が最大となる m = m_optを探す*/
        a_max = 0.0;
        for(m=k;m<N;m++){
            if(fabs(a[pivot[m]][k])>a_max){<br />
                a_max = fabs(a[pivot[m]][k]);<br />
                m_opt = m;<br />
            }<br />
        }<br />
        /*m_opt != kのとき<br />
          「k段のピポット」と「m_opt段のピポット」を交換*/<br />
        if(m_opt != k){<br />
            k_step_old = pivot[k];<br />
            pivot[k] = pivot[m_opt];<br />
            pivot[m_opt] = k_step_old;<br />
        }<br />
        //前進消去<br />
        for(m=k+1;m<N;m++){
                ell = a[pivot[m]][k]/a[pivot[k]][k];

                for(j=k+1;j<N;j++){
                    a[pivot[m]][j] = a[pivot[m]][j] - ell*a[pivot[k]][j];
                }
                b[pivot[m]] = b[pivot[m]] - ell*b[pivot[k]];
        }
    }
    //後退代入
    //b[pivot[0]],b[pivot[1]],....に解が代入される
    for(m=N-1;m>=0;m--){<br />
        for(j=N-1;j>m;j--){<br />
            b[pivot[m]] =  b[pivot[m]] -a[pivot[m]][j]*b[pivot[j]];<br />
        }<br />
        b[pivot[m]] =  b[pivot[m]]/a[pivot[m]][m];<br />
    }<br />
    <br />
    printf("%f\n",b[pivot[0]]);<br />
    printf("%f\n",b[pivot[1]]);<br />
    printf("%f\n",b[pivot[2]]);<br />
    printf("%f\n",b[pivot[3]]);<br />
}</pre></p>
