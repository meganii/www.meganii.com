    <h4>FlickrRIA.<a class="keyword" href="http://d.hatena.ne.jp/keyword/mxml">mxml</a></h4>
    <pre class="code" data-lang="" data-unlink>&lt;?xml version=&#34;1.0&#34; encoding=&#34;utf-8&#34;?&gt;
&lt;mx:Application xmlns:mx=&#34;http://www.adobe.com/2006/mxml&#34;
    backgroundGradientColors=&#34;[0xFFFFFF, 0xAAAAAA]&#34;
    horizontalAlign=&#34;left&#34;
    verticalGap=&#34;15&#34;
    horizontalGap=&#34;15&#34;&gt;

    &lt;mx:Script&gt;
        &lt;![CDATA[
            import mx.collections.ArrayCollection;
            import mx.rpc.events.ResultEvent;

            [Bindable]
            private var photoFeed:ArrayCollection;

            private function requestPhotos():void {
                photoService.cancel();
                var params:Object = new Object();
                params.format = &#39;rss_200_enc&#39;;
                params.tags = searchTerms.text;
                photoService.send(params);
            }

            private function photoHandler(event:ResultEvent):void {
                 photoFeed = event.result.rss.channel.item as ArrayCollection;
            }
         ]]&gt;
    &lt;/mx:Script&gt;

    &lt;mx:HTTPService id=&#34;photoService&#34;
        url=&#34;http://api.flickr.com/services/feeds/photos_public.gne?id=hogehoge@hoge&#34;
        result=&#34;photoHandler(event)&#34; /&gt;

	&lt;mx:HBox&gt;
		&lt;mx:Label text=&#34;Flickr tags or search terms:&#34; /&gt;
		&lt;mx:TextInput id=&#34;searchTerms&#34; /&gt;
		&lt;mx:Button label=&#34;Search&#34;
			click=&#34;requestPhotos()&#34; /&gt;
	&lt;/mx:HBox&gt;

	&lt;mx:TileList width=&#34;100%&#34; height=&#34;100%&#34;
		dataProvider=&#34;{photoFeed}&#34;
		itemRenderer=&#34;FlickrThumbnail&#34;&gt;
	&lt;/mx:TileList&gt;

&lt;/mx:Application&gt;</pre>
